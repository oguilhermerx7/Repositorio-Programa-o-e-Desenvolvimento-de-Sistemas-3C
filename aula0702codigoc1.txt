#include <stdio.h>

int main() {
    int tabuada, resultado, contador;

    printf("Digite a tabuada desejada: ");
    scanf("%d", &tabuada);

    for (contador = 1; contador <= 10; contador++) {
        resultado = tabuada * contador;
        printf("%d * %d = %d\n", tabuada, contador, resultado);
    }

    return 0;
}
